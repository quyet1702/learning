Thuộc tính vue trên html(template):
	ví dụ: <p v-if="true">Now you see me</p>
	v-if="true"					// nếu trả về false sẽ remove thẻ; nếu true thì sẽ hiện thẻ
	v-html="var"				// lấy giá trị var gắn vào nội dung của thẻ
	:href = "url"				// nhận giá trị hoặc biến chứa giá trị url
	v-model="propertyName"		// sử dụng để cập nhật giá trị của input vào propertyName mỗi khi giá trị thay đổi
	:disabled="isDisable"		// nhận giá trị true false; nếu = true thì sẽ disabled và ngược lại
	:id="propertyName"			// Tạo thuộc tính id và gắn giá trị cho thuộc tính; giá trị này sẽ thay đổi nếu property gắn thay đổi
	@click=""
	v-bind:class="[propertyName, propertyName]"									// thêm class bằng cách lấy giá trị của property
	v-bind:class="{ active: propertyName, 'text-danger': propertyName }"		// thêm class active nếu propertyName trả về true
	v-bind:class="[{ active: isActive }, errorClass]"							// thêm class theo điều kiện
	v-bind:style="{ color: propertyName, fontSize: propertyName + 'px' }"		// in các giá trị dưới dạng chuỗi của propertyName để tạo style
	v-bind:style="styleObject"													// data: {
																						styleObject: {
																							color: 'red',
																							fontSize: '13px'
																						}
																						}
	v-bind:style="[baseStyles, overridingStyles]"								// sử dụng nhiều style object
	v-bind:style="{ display: ['-webkit-box', '-ms-flexbox', 'flex'] }"			// in ra 1 trong các giá trị mà trình duyệt hỗ trợ
	v-on:click="doSomething"
	v-show="true"																// thêm thuộc tính display none khi v-show="false" và display="block" khi v-show="true"
	@[eventName]="doSomething"
	:data-name="methodName(data)"	// có thể gắn giá trị trả về của method cho thuộc tính
	<li v-for="varName in property">{{ varName.text }}</li>		// tạo vòng lặp duyệt qua từng phần tử của property và mỗi phần tử in thẻ li chứa nó
	<input v-model="message">		// giá trị của input sẽ ngay lập tức cập nhật vào property message

		Nếu muốn tạo mới thẻ thì hãy dùng thuộc tính key="abc"	và gắn chuỗi bất kỳ để báo cho Vue là 2 thẻ này khác nhau và đừng dùng lại

Cấu hình VueJS:
		export default {
			data() {
				return {
					author: {
						name: 'John Doe',
						books: [
						'Vue 2 - Advanced Guide',
						'Vue 3 - Basic Guide',
						'Vue 4 - The Mystery'
						]
					}
				}
			},
			computed: {
				// a computed getter
				publishedBooksMessage() {							// thực hiện các chức năng logic sau đó trả về kết quả
					// `this` points to the component instance
					return this.author.books.length > 0 ? 'Yes' : 'No'
				}
			},
			methods: {												// cách 2: sử dụng method thay vì hàm
				calculateBooksMessage() {
					return this.author.books.length > 0 ? 'Yes' : 'No'
				}
			}
		}				
		<span>{{ publishedBooksMessage }}</span>
		<span>{{ calculateBooksMessage() }}</span>
