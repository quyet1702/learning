Regexp:
	cú pháp: /pattern/modifier(s);
		Modifiers:
			g		// Tìm kiếm toàn cầu

			[abc]		// Đại diện cho 1 hoặc nhiều chữ bất kỳ
			[a-z]		// chữ thường
			[A-z]		// chữ thường hoặc in hoa
			[^abc]		// Tìm bất kỳ ký tự nào không nằm trong dấu ngoặc
			[0-9]		// Tìm bất kỳ số nào trong dấu ngoặc
			[^0-9]		// Tìm bất kỳ số nào Không nằm trong dấu ngoặc
			(x|y)		// Tìm bất kỳ ký tự nào là x hoặc y
			\w+		// Tìm kiếm 1 từ có ít nhất 1 ký tự
			^abc		// Tìm kiếm khớp ở đầu dòng
			abc$		// Tìm kiếm khớp ở cuối dòng
			a.c		// Tìm kiếm dấu . có thể là bất kỳ ký tự nào
			\w		// là số chữ và _
			\W		// Không phải số chữ và _
			\d		// là số 0->9
			\D		// Không phải là số 0->9
			\s		// là khoảng trắng
			\S		// không phải là khoảng trắng
			\n		// ký tự xuống dòng
			\t		// ký tự tab	
			\bABC		// Tìm kiếm ABC ở đầu của 1 từ
			ABC\b		// Tìm kiếm ABC ở cuối của 1 từ
			\BABC		// Tìm kiếm ABC Không ở đầu của 1 từ
			ABC\B		// Tìm kiếm ABC Không ở cuối của 1 từ
			X{4}		// X có 4 ký tự
			X{3,4}		// X có 3 đến 4 ký tự
			X{3,}		// X có ít nhất 3 ký tự
			is(?=all)	// Tìm kiếm 'is' mà trong chuỗi có 'all' theo sau
			is(?!all)	// Tìm kiếm 'is' mà trong chuỗi không có 'all' theo sau
			XB*		// Tìm kiếm ký tự A mà theo sau là 0 hoặc nhiều ký tự B
			X+		// X có ít nhất 1 từ
			X? 		// X có thể có hoặc không có =  X{0,1}
			()		// các phép tìm kiếm trong dấu ngoặc sẽ trả về chuỗi tìm thấy
