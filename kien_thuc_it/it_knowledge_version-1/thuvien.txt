	login:
		luồng dữ liệu:
		-> chuyển hướng trang login nếu chưa login sử dụng middleware (App\Http\Middleware\Authenticate.php -> route\auth.php -> App\Http\Controllers\Auth\AuthenticatedSessionController.create -> view\auth\login)
		-> khi login thì đối chiếu với db rồi lưu session (route\auth -> App\Http\Controllers\Auth\AuthenticatedSessionController.store -> App\Http\Requests\Auth\LoginRequest -> App\Providers\RouteServiceProvider -> route\web)
		-> khi đã login sẽ cấp đối tượng chứa dữ liệu user (đối tượng đã đc inject chứa dữ liệu user  {{Auth::user() ->name}} {{Auth::user() ->email}} {{Auth::user()->created_at}}...)
		-> chỉnh sửa thông tin user (route/web -> ProfileController.update -> App\Http\Controllers\ProfileController)
		-> xóa user (route/web -> ProfileController.destroy -> App\Http\Controllers\ProfileController)
		-> xác nhận mật khẩu (route\auth\confirm-password -> ConfirmablePasswordController.show -> view -> ConfirmablePasswordController.store)
		-> đăng xuất (route\auth\logout -> AuthenticatedSessionController.destroy -> view)

	tạo model+migrate:		php artisan make:model Role -m
	thêm cột cho migrate:		$table->string('name');
	thêm ràng buộc cho Role:	public function Users(){
        					return $this->hasMany(User::class);
    					}
	thêm cột role_id cho User từ migrate:
					$table->integer('role_id');
	refresh lại db:			php artisan migrate hoặc php artisan migrate:refresh	nếu đã có table trước đó
	thêm seeder:			php artisan make:seeder roleSeeder
	sửa file roleSeeder method run:	
					$role = new Role();
        				$role->name = 'admin';
        				$role->save();
	khai báo trên databaseSeeder:	$this->call(roleSeeder::class);
	thêm ràng buộc cho User:
   					public function Role(){
        					return $this->belongsTo(Role::class);
    					}
	thêm role_id mặc định cho User khi tạo user:

        					'role_id' => 1,
    					];
	thêm middleware:
		php artisan make:middleware RoleMiddleware
		sửa middleware RoleMiddleware:
			public function handle(Request $request, Closure $next,$role)
    			{
        			if(!(Auth::user() || Auth::user()->role->name==$role)){
            				return redirect()->route('login');
        			}
        			return $next($request);
    			}
		đăng kí middleware app/http/kernel.php: 	'role' => \App\Http\Middleware\RoleMiddleware::class,
		sử dụng trên route:

					// trong đó: role là tên đã đăng kí trên kernel; admin là tham số $role sẽ đc truyền vô trong Middleware đã tạo
