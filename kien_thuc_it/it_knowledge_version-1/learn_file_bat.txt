	move "test.bat" "folder"		-> di chuyển file test.bat vào folder

	copy "fileName1" "folderName2"

		-> nếu fileName1 là 1 folder thì sẽ biến folderName2 giống fileName1
		-> nếu folderName2 là folder thì lệnh ko có tác dụng
	rename "txt1.bat" "test.bat"	-> đổi tên file từ txt1.bat thành test.bat
		-> có thể viết ren thay vì rename

	Config file:
		attrib		-> hiện thị các file và thuộc tính của file
		file chỉ đc phép đọc:	attrib +r fileName
			xóa thuộc tính: attrib -r fileName
		ẩn file: attrib -h list.txt		-> không thể nhìn thấy 
		hiện file: attrib -h list.txt
	Lấy dữ liệu từ file:
		 FOR /F "tokens=* delims=" %%x in (a.txt) DO set b=%%x	-> lấy dữ liệu từ file a.txt và gắn vào biến b
	Gắn dữ liệu vào file:
		@echo 123> test1.txt	xóa nội dung cũ và chèn text
		@echo 123>> test1.txt	chèn vào dòng mới

	cd folderName		-> truy cập vào folder con
	mrdir			-> xóa folder
	dir "Program Files"	-> kiểm tra thư mục con và file của Program Files

	for %%b in ("%cd%") do set project=%%~nxb	-> lấy folder chứa file thực thi
