Làm việc với commit:
	Thêm:
		git add filename						hoặc git add .
		git commit -m "sdf"
		git mv oldName newName					// đổi tên folder
	Xóa:
		xóa commit:
			git reset --soft HEAD~1				// xóa commit mới nhất nhưng ko xóa những thay đổi
			git reset --hard HEAD~1				// xóa commit mới nhất và xóa những thay đổi
		xóa thay đổi:
			xóa bỏ những thay đổi ở file untracked so với commit cuối:
				git restore filename			// hoặc 	git restore .
			thay đổi trạng thái staged -> về untracked:
				git restore --staged file_name		// hoặc git restore --staged .
		
		Xóa tệp không được theo dõi:
			git clean -fd			// xóa những tệp và thư mục
			git clean -f			// chỉ xóa những tệp
	Sửa:
		git commit --amend -m "new massage"			// sửa nội dung thay đổi hoặc massage của commit cuối
		git checkout commitCode -- file_name			// lấy file của commit cũ;
									// không nên sử dụng git checkout commitCode -- vì nó sẽ tạo head detach; và tách ra khỏi branch
	kiểm tra							// bấm lại git check branName để khắc phục head detach
		git diff --staged filename				// kiểm tra file staged với commit hiện tại
		git diff						// so sánh từng file làm việc với staged mới nhất; nếu ko có staged thì so sánh với commit gần nhất 
		git diff mãCommit1 mãcommit2 > filename			// khiểm tra sự khác nhau giữa 2 commit	
		git log --oneline					// xem tổng quan các commit
		git log --oneline --graph				// xem tổng quan kèm rẽ nhánh
	
	bỏ qua theo dõi:
		tạo file: .gitignore
		thêm nội dung file:	# Visual Studio Code		comment
					*.tmp	bỏ qua các file có đuôi .tmp
					folderName 			bỏ qua các folder chỉ định
					path/folderName
					[Bb]in				Bin hoặc bin
	git stash:
		- Lưu tạm thời những thay đổi
		git stash		// chuyển những thay đổi từ thư mục làm việc vào stash và xóa sạch những modified
		git stash apply		// đưa những thay đổi từ stash vào thư mục làm việc
		git stash pop		// đưa những thay đổi từ stash vào thư mục làm việc và làm sạch stash
		git stash clear		// xóa toàn bộ stash

Làm việc với branch:
	Thêm:
		git branch newBranch
		git checkout -b BranchName		// thêm mới và chuyển qua nhánh
	xóa:
		git branch -d BranchName		// chỉ xóa đc khi nhánh này đã đc merge
		git branch -D branchName		// xóa nhánh không cần merge
	chuyển:
		git checkout BranchName
	
	Đổi tên branch:
		git branch -m old_branch new_branch
	Xóa commit trên repository:
		bước 1: git reset --soft HEAD~N			N là số commit muốn xóa
		bước 2: sửa lại nội dung muốn sửa nếu có
		bước 3: tạo 1 commit và push lên repo
	gộp nhánh:
		git merge ten_nhanh_can_gop		// nhánh chính sẽ lấy hết commit của nhánh cần gộp
	hủy bỏ xung đột:
		git reset --merge
	sử lý xung đột khi gộp nhánh:
		khi gộp nhánh xuất hiện thông báo: CONFLICT (content): Merge conflict in filename
		bước 1: git mergetool
		bước 2: nhấn enter
		bước 3: 
			nhấn :diffg LO		để lấy thay đổi của master(bên trái)
			nhấn :diffg RE		để lấy thay đổi của nhánh cần gộp (bên phải)
			nhấn :diffg BA		để hủy bỏ thay đổi
	Giải quyết conflicts:
		bước 1: chuyển nhánh phụ bị dính conflicts
		bước 2: pull với nhánh chính
		bước 3: xem chỗ bị dính conflict: màu xanh lá là chọn phần của nhánh phụ bị dính conflicts
								màu xanh dương là chọn phần của nhánh chính bị xung đột
		bước 4: git add file_chứa_conflicts
		bước 5: git commit -m "sdf"
		bước 6 git push
	kiểm tra:
		git branch				// kiểm tra branch trên local
		git branch -a			// kiểm tra branch trên server

Làm việc với repository:
	Kiểm tra:
		git remote -v			// kiểm tra các remote
	kết nối:
		git remote add origin URL
	lấy resource về:
		git clone url folderName
		git clone --branch <branchname> <remote-repo-url>	// lấy resource từ 1 nhánh củ thê
	lấy thay đổi từ nhánh github về local:
		git pull origin master
	gửi thay đổi từ nhánh local lên github:
		git push origin master
	tạo nhánh repository
		git checkout -b branchName		// tạo nhánh local
		git push origin branchName		// push lên đồng thời tự tạo nhánh
	Cài đặt nhánh mặc định:
		git push --set-upstream origin master
	Thay đổi url git:
		git remote set-url origin <new_url>
	đổi tên remote
		git remote rename oldName newName
	xóa brand ở server: git push origin --delete alpha
	Tăng giới hạn:
		git config http.postBuffer 524288000	// tăng giới hạn dữ liệu đc đẩy lên github lên tới 500MB
	submodule:

Đẩy lên:
	git push origin master			//chú ý tên branch trên local phải trùng với branch trên rep
	Đẩy lên với nhánh khác
	1. Pull code từ nhánh gốc: git pull origin develop
	2. Tạo nhánh mới: git checkout -b + tên nhánh mới
	3. Add file vào git: git add . / git add "file"
	4. Commit code: git commit -m "message commit"
	5. Tạo nhánh mới: git checkout -b bugfix/4254q
	5. Lần đầu push: git push --set-upstream origin + tên nhánh mới
	6. Sau lần đầu tiên push: git push origin + tên nhánh
		$ git checkout -b test-branch 56a4e5c08
