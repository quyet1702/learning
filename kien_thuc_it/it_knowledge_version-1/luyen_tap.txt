Lưu file và public lên:
    static function addFile($file, $folder = 'upload', array $exceptFileName ):string
    {
        // check trùng
        $imagePath = 'public/'.$folder;

        // Lưu file ảnh
        $fileImage = $file;
        $imageName = $file->getClientOriginalName();

        // Tạo tên file ảnh không trùng trong db
        do {
            $imageName = 'song'.self::random().'.'.$file->getClientOriginalExtension();
        } while(in_array('storage/'.$folder.'/'.$imageName, $exceptFileName));

        // Lưu ảnh vào storage
        Storage::putFileAs($imagePath, $fileImage, $imageName);
        return 'storage/'.$folder.'/'.$imageName;
    }
